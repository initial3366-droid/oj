### 3. 实现的核心功能

#### ✅ 自动 Token 管理

```typescript
// 自动获取 token
function getToken(): string | null {
  return window.localStorage.getItem("qoj.accessToken");
}

// 自动保存 token
export function setToken(token: string) {
  window.localStorage.setItem("qoj.accessToken", token);
}

// 自动清除 token
export function clearToken() {
  window.localStorage.removeItem("qoj.accessToken");
}

// 请求自动带 token
const response = await fetch(url, {
  headers: token ? { Authorization: `Bearer ${token}` } : undefined,
});
```

#### ✅ 统一错误处理

```typescript
// 401 未授权
if (response.status === 401) {
  clearToken();
  if (window.location.pathname !== "/auth") {
    window.location.href = "/auth";
  }
  throw new Error("未登录或登录已过期，请重新登录");
}

// 403 禁止访问
if (response.status === 403) {
  throw new Error(body?.message || "权限不足，无法访问该资源");
}

// 500 服务器错误
if (response.status === 500) {
  throw new Error(body?.message || "服务器内部错误，请稍后重试");
}
```

#### ✅ 统一响应解析

```typescript
async function parseResponse<T>(response: Response): Promise<T> {
  let body: ApiResponse<T> | null = null;
  
  try {
    body = (await response.json()) as ApiResponse<T>;
  } catch {
    body = null;
  }
  
  // 错误处理...
  
  // 验证响应格式
  if (!body || body.code !== 200) {
    throw new Error(body?.message || "请求失败");
  }
  
  return body.data;
}
```

#### ✅ 超时保护

```typescript
const API_TIMEOUT_MS = 15000;

function timeoutSignal() {
  const controller = new AbortController();
  const timeout = window.setTimeout(() => controller.abort(), API_TIMEOUT_MS);
  return { controller, timeout };
}

async function fetchWithTimeout(url: string, init: RequestInit) {
  const { controller, timeout } = timeoutSignal();
  try {
    return await fetch(url, {
      ...init,
      signal: controller.signal,
    });
  } catch (error) {
    if (error instanceof DOMException && error.name === "AbortError") {
      throw new Error("请求超时，请检查后端服务");
    }
    throw error;
  } finally {
    window.clearTimeout(timeout);
  }
}
```
